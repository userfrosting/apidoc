 13 /**
 14  * MessageTranslator Class
 15  *
 16  * Translate message ids to a message in a specified language.
 17  *
 18  * @author    Louis Charette
 19  * @author    Alexander Weissman (https://alexanderweissman.com)
 20  */
 21 class MessageTranslator extends Repository
 22 {
 23     /**
 24      * @var Twig_Environment A Twig environment used to replace placeholders.
 25      */
 26     protected $twig;
 27 
 28     /**
 29      * @var string The default key that contains the pluralization code.
 30      */
 31     protected $defaultPluralKey = 'plural';
 32 
 33     /**
 34      * Create the translator.
 35      *
 36      * @param array $items
 37      */
 38     public function __construct(array $items = [])
 39     {
 40         $this->items = $items;
 41 
 42         $loader = new \Twig_Loader_Filesystem();
 43         $this->twig = new \Twig_Environment($loader);
 44     }
 45 
 46     /**
 47      * Translate the given message id into the currently configured language, substituting any placeholders that appear in the translated string.
 48      *
 49      * Return the $messageKey if not match is found
 50      * @param string $messageKey The id of the message id to translate. can use dot notation for array
 51      * @param array|int $placeholders[optional] An optional hash of placeholder names => placeholder values to substitute.
 52      * @return string The translated message.
 53      */
 54     public function translate($messageKey, $placeholders = [])
 55     {
 56         // If we didn't find a match, we simply apply the placeholders to $messageKey
 57         if (!$this->has($messageKey)) {
 58             $message = $messageKey;
 59         } else {
 60             // Get the message
 61             $message = $this->get($messageKey);
 62 
 63             /* If the message is an array, we have to go deeper because an array can countain some special handles:
 64                 - @TRANSLATION
 65                 - @REPLACE
 66                 - @TRANSLATE
 67                 - @PLURAL
 68             */
 69             if (is_array($message)) {
 70                 // Is the message array countain any plural rules (keys that are int)
 71                 if (!empty(array_filter(array_keys($message), 'is_int'))) {
 72 
 73                     // Now we can handle plurals using the @PLURAL to define the plural key. If it's not defined, we use the default one
 74                     $pluralKey = (isset($message['@PLURAL'])) ? $message['@PLURAL'] : $this->defaultPluralKey;
 75 
 76                     // We try get the plural value and default to `1` if none is found
 77                     // We also allow for a shortcut using the second argument as a numeric value for simple strings.
 78                     $pluralValue = (isset($placeholders[$pluralKey]) ? (int) $placeholders[$pluralKey] : (!is_array($placeholders) && is_numeric($placeholders) ? $placeholders : null));
 79 
 80                     // Stop for a sec... We don't have a plural value, but before defaut to 1, we check if there's any @TRANSLATION handle
 81                     if (is_null($pluralValue) && (!$this->has($messageKey . ".@TRANSLATION") || $this->get($messageKey . ".@TRANSLATION") == null)) {
 82 
 83                         //Default
 84                         $pluralValue = 1;
 85 
 86                     }
 87 
 88                     // If plural value is still null, we have found our message..!
 89                     if (is_null($pluralValue)) {
 90                         $message = $this->get($messageKey . ".@TRANSLATION");
 91                     } else {
 92 
 93                         // Ok great. Now we need the right plural form.
 94                         // N.B.: Plurals is based on phpBB and Mozilla work : https://developer.mozilla.org/en-US/docs/Mozilla/Localization/Localization_and_Plurals
 95                         $keyFound = false;
 96 
 97                         // 0 is handled differently. We use it so that "0 users" may be displayed as "No users".
 98                         if ($pluralValue == 0 && isset($message[0])) {
 99                             $keyFound = 0;
100                         } else {
101 
102                             $usePluralForm = $this->getPluralForm($pluralValue);
103                             if (isset($message[$usePluralForm])) {
104                                 // The key we need exists, so we use it.
105                                 $keyFound = $usePluralForm;
106                             } else {
107                                 // If the key we need doesn't exist, we use the previous one.
108                                 $numbers = array_keys($message);
109                                 foreach ($numbers as $num) {
110                                     if (is_int($num) && $num > $usePluralForm) {
111                                         break;
112                                     }
113                                     $keyFound = $num;
114                                 }
115                             }
116                         }
117 
118                         // If no key was found, use the last entry (because it is mostly the plural form).
119                         if ($keyFound === false) {
120                             $numbers = array_keys($message);
121                             $keyFound = end($numbers);
122                         }
123 
124                         $message = $message[$keyFound];
125 
126                         // If we used the shortcut and $placeholders is a numeric value
127                         // it must be passed back as an array for replacement in the main $message
128                         if (is_numeric($placeholders) || empty($placeholders)) {
129                             $placeholders = array($pluralKey => $pluralValue);
130                         }
131                     }
132 
133                 // @TRANSLATION => When $messageKey is an array, this key is used. To use this, we can't have a plural value
134                 } elseif ($this->has($messageKey . ".@TRANSLATION")) {
135                     $message = $this->get($messageKey . ".@TRANSLATION");
136                 // If we don't have plural AND a @TRANSLATION, we can't translate any translation key, so we will simply apply the placeholders to $messageKey
137                 } else {
138                     $message = $messageKey;
139                 }
140             }
141         }
142 
143         // Ok, now we have a $message and need to replace the placeholders
144         // Make sure $placeholders is an array otherwise foreach will fail
145         if (!is_array($placeholders)) {
146             return $message;
147         }
148 
149         // Interpolate translatable placeholders values. This allows to
150         // pre-translate placeholder which value starts with the `&` caracter
151         foreach ($placeholders as $name => $value) {
152             //We don't allow nested placeholders. They will return errors on the next lines
153             if (is_array($value)) {
154                 continue;
155             }
156 
157             // We test if the placeholder value starts the "&" caracter.
158             // That means we need to translate that placeholder value
159             if (substr($value, 0, 1) === '&') {
160                 // Remove the current placeholder from the master $placeholder
161                 // array, otherwise we end up in an infinite loop
162                 $data = array_diff($placeholders, [$name => $value]);
163 
164                 // Translate placeholders value and place it in the main $placeholder array
165                 $placeholders[$name] = $this->translate(ltrim($value, '&'), $data);
166             }
167         }
168 
169         // We check for {{&...}} strings in the resulting message.
170         // While the previous loop pre-translated placeholder value, this one
171         // pre-translate the message string vars
172         // We use some regex magic to detect them !
173         $message = preg_replace_callback("/{{&(([^}]+[^a-z]))}}/", function ($matches) use ($placeholders) {
174             return $this->translate($matches[1], $placeholders);
175         }, $message);
176 
177         // Now it's time to replace the remaining placeholder. We use Twig do to this.
178         // It's a bit slower, but allows to use the many Twig filters
179         // See: http://twig.sensiolabs.org/doc/2.x/
180         $template = $this->twig->createTemplate($message);
181         $message = $template->render($placeholders);
182 
183         // Done !
184         return $message;
185     }
186 
187     /**
188     * Determine which plural form we should use.
189     * For some languages this is not as simple as for English.
190     *
191     * @param $number        int|float   The number we want to get the plural case for. Float numbers are floored.
192     * @param $forceRule    mixed   False to use the plural rule of the language package
193     *                               or an integer to force a certain plural rule
194     * @return   int     The plural-case we need to use for the number plural-rule combination
195     */
196     public function getPluralForm($number, $forceRule = false)
197     {
198         $number = (int) $number;
199 
200         // Default to English rule (1) or the forced one
201         $rule = ($forceRule !== false) ? $forceRule : (($this->has('@PLURAL_RULE')) ? $this->get('@PLURAL_RULE') : 1);
202 
203         if ($rule > 15 || $rule < 0) {
204             throw new OutOfRangeException("The rule number '$rule' must be between 0 and 16.");
205         }
206 
207         /**
208         * The following plural rules are based on a list published by the Mozilla Developer Network & code from phpBB Group
209         * https://developer.mozilla.org/en-US/docs/Mozilla/Localization/Localization_and_Plurals
210         */
211         switch ($rule) {
212             case 0:
213                 /**
214                 * Families: Asian (Chinese, Japanese, Korean, Vietnamese), Persian, Turkic/Altaic (Turkish), Thai, Lao
215                 * 1 - everything: 0, 1, 2, ...
216                 */
217                 return 1;
218             case 1:
219                 /**
220                 * Families: Germanic (Danish, Dutch, English, Faroese, Frisian, German, Norwegian, Swedish), Finno-Ugric (Estonian, Finnish, Hungarian), Language isolate (Basque), Latin/Greek (Greek), Semitic (Hebrew), Romanic (Italian, Portuguese, Spanish, Catalan)
221                 * 1 - 1
222                 * 2 - everything else: 0, 2, 3, ...
223                 */
224                 return ($number == 1) ? 1 : 2;
225             case 2:
226                 /**
227                 * Families: Romanic (French, Brazilian Portuguese)
228                 * 1 - 0, 1
229                 * 2 - everything else: 2, 3, ...
230                 */
231                 return (($number == 0) || ($number == 1)) ? 1 : 2;
232             case 3:
233                 /**
234                 * Families: Baltic (Latvian)
235                 * 1 - 0
236                 * 2 - ends in 1, not 11: 1, 21, ... 101, 121, ...
237                 * 3 - everything else: 2, 3, ... 10, 11, 12, ... 20, 22, ...
238                 */
239                 return ($number == 0) ? 1 : ((($number % 10 == 1) && ($number % 100 != 11)) ? 2 : 3);
240             case 4:
241                 /**
242                 * Families: Celtic (Scottish Gaelic)
243                 * 1 - is 1 or 11: 1, 11
244                 * 2 - is 2 or 12: 2, 12
245                 * 3 - others between 3 and 19: 3, 4, ... 10, 13, ... 18, 19
246                 * 4 - everything else: 0, 20, 21, ...
247                 */
248                 return ($number == 1 || $number == 11) ? 1 : (($number == 2 || $number == 12) ? 2 : (($number >= 3 && $number <= 19) ? 3 : 4));
249             case 5:
250                 /**
251                 * Families: Romanic (Romanian)
252                 * 1 - 1
253                 * 2 - is 0 or ends in 01-19: 0, 2, 3, ... 19, 101, 102, ... 119, 201, ...
254                 * 3 - everything else: 20, 21, ...
255                 */
256                 return ($number == 1) ? 1 : ((($number == 0) || (($number % 100 > 0) && ($number % 100 < 20))) ? 2 : 3);
257             case 6:
258                 /**
259                 * Families: Baltic (Lithuanian)
260                 * 1 - ends in 1, not 11: 1, 21, 31, ... 101, 121, ...
261                 * 2 - ends in 0 or ends in 10-20: 0, 10, 11, 12, ... 19, 20, 30, 40, ...
262                 * 3 - everything else: 2, 3, ... 8, 9, 22, 23, ... 29, 32, 33, ...
