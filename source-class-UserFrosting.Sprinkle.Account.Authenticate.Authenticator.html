 81     /**
 82      * Create a new Authenticator object.
 83      *
 84      * @param ClassMapper $classMapper Maps generic class identifiers to specific class names.
 85      * @param Session $session The session wrapper object that will store the user's id.
 86      * @param Config $config Config object that contains authentication settings.
 87      */
 88     public function __construct(ClassMapper $classMapper, Session $session, $config, $cache)
 89     {
 90         $this->classMapper = $classMapper;
 91         $this->session = $session;
 92         $this->config = $config;
 93         $this->cache = $cache;
 94 
 95         // Initialize RememberMe storage
 96         $this->rememberMeStorage = new RememberMePDO($this->config['remember_me.table']);
 97     
 98         // Get the actual PDO instance from Eloquent
 99         $pdo = Capsule::connection()->getPdo();
100 
101         $this->rememberMeStorage->setConnection($pdo);
102 
103         // Set up RememberMe
104         $this->rememberMe = new RememberMe($this->rememberMeStorage);
105         // Set cookie name
106         $cookieName = $this->config['session.name'] . '-' . $this->config['remember_me.cookie.name'];
107         $this->rememberMe->getCookie()->setName($cookieName);
108 
109         // Change cookie path
110         $this->rememberMe->getCookie()->setPath($this->config['remember_me.session.path']);
111 
112         // Set expire time, if specified
113         if ($this->config->has('remember_me.expire_time') && ($this->config->has('remember_me.expire_time') != null)) {
114             $this->rememberMe->getCookie()->setExpireTime($this->config['remember_me.expire_time']);
115         }
116 
117         $this->user = null;
118 
119         $this->viaRemember = false;
120     }

122     /**
123      * Attempts to authenticate a user based on a supplied identity and password.
124      *
125      * If successful, the user's id is stored in session.
126      */
127     public function attempt($identityColumn, $identityValue, $password, $rememberMe = false)
128     {
129         // Try to load the user, using the specified conditions
130         $user = $this->classMapper->staticMethod('user', 'where', $identityColumn, $identityValue)->first();
131 
132         if (!$user) {
133             throw new InvalidCredentialsException();
134         }
135 
136         // Check that the user has a password set (so, rule out newly created accounts without a password)
137         if (!$user->password) {
138             throw new InvalidCredentialsException();
139         }
140 
141         // Check that the user's account is enabled
142         if ($user->flag_enabled == 0) {
143             throw new AccountDisabledException();
144         }
145 
146         // Check that the user's account is activated
147         if ($user->flag_verified == 0) {
148             throw new AccountNotVerifiedException();
149         }
150 
151         // Here is my password.  May I please assume the identify of this user now?
152         if (Password::verify($password, $user->password)) {
153             $this->login($user, $rememberMe);
154             return $user;
155         } else {
156             // We know the password is at fault here (as opposed to the identity), but lets not give away the combination in case of someone bruteforcing
157             throw new InvalidCredentialsException();
158         }
159     }

181     /**
182      * Process an account login request.
183      *
184      * This method logs in the specified user, allowing the client to assume the user's identity for the duration of the session.
185      * @param User $user The user to log in.
186      * @param bool $rememberMe Set to true to make this a "persistent session", i.e. one that will re-login even after the session expires.
187      * @todo Figure out a way to update the currentUser service to reflect the logged-in user *immediately* in the service provider.
188      * As it stands, the currentUser service will still reflect a "guest user" for the remainder of the request.
189      */
190     public function login($user, $rememberMe = false)
191     {
192         $oldId = session_id();
193         $this->session->regenerateId(true);
194 
195         // Since regenerateId delete the old session, we'll do the same in cache
196         $this->cache->tags("_s".$oldId)->flush();
197 
198         // If the user wants to be remembered, create Rememberme cookie
199         if ($rememberMe) {
200             $this->rememberMe->createCookie($user->id);
201         } else {
202             $this->rememberMe->clearCookie();
203         }
204 
205         // Assume identity
206         $key = $this->config['session.keys.current_user_id'];
207         $this->session[$key] = $user->id;
208 
209         // Set auth mode
210         $this->viaRemember = false;
211 
212         // User login actions
213         $user->onLogin();
214     }

216     /**
217      * Processes an account logout request.
218      *
219      * Logs the currently authenticated user out, destroying the PHP session and clearing the persistent session.
220      * This can optionally remove persistent sessions across all browsers/devices, since there can be a "RememberMe" cookie
221      * and corresponding database entries in multiple browsers/devices.  See http://jaspan.com/improved_persistent_login_cookie_best_practice.
222      *
223      * @param bool $complete If set to true, will ensure that the user is logged out from *all* browsers on all devices.
224      */
225     public function logout($complete = false)
226     {
227         $currentUserId = $this->session->get($this->config['session.keys.current_user_id']);
228 
229         // This removes all of the user's persistent logins from the database
230         if ($complete) {
231             $this->storage->cleanAllTriplets($currentUserId);
232         }
233 
234         // Clear the rememberMe cookie
235         $this->rememberMe->clearCookie();
236 
237         // User logout actions
238         if ($currentUserId) {
239             $currentUser = $this->classMapper->staticMethod('user', 'find', $currentUserId);
240             if ($currentUser) {
241                 $currentUser->onLogout();
242             }
243         }
244 
245         $this->user = null;
246         $this->loggedOut = true;
247 
248         $oldId = session_id();
249 
250         // Completely destroy the session
251         $this->session->destroy();
252 
253         // Since regenerateId delete the old session, we'll do the same in cache
254         $this->cache->tags("_s".$oldId)->flush();
255 
256         // Restart the session service
257         $this->session->start();
258     }

260     /**
261      * Try to get the currently authenticated user, returning a guest user if none was found.
262      *
263      * Tries to re-establish a session for "remember-me" users who have been logged out due to an expired session.
264      * @return User|null
265      * @throws AuthExpiredException
266      * @throws AuthCompromisedException
267      * @throws AccountInvalidException
268      * @throws AccountDisabledException
269      */
270     public function user()
271     {
272         $user = null;
273 
274         if (!$this->loggedOut) {
275 
276             // Return any cached user
277             if (!is_null($this->user)) {
278                 return $this->user;
279             }
280 
281             // If this throws a PDOException we catch it and return null than allowing the exception to propagate.
282             // This is because the error handler relies on Twig, which relies on a Twig Extension, which relies on the global current_user variable.
283             // So, we really don't want this method to throw any database exceptions.
284             try {
285                 // Now, check to see if we have a user in session
286                 $user = $this->loginSessionUser();
287 
288                 // If no user was found in the session, try to login via RememberMe cookie
289                 if (!$user) {
290                     $user = $this->loginRememberedUser();
291                 }
292             } catch (\PDOException $e) {
293                 $user = null;
294             }
295         }
296 
297         return $this->user = $user;
298     }

311     /**
312      * Attempt to log in the client from their rememberMe token (in their cookie).
313      *
314      * @return User|bool If successful, the User object of the remembered user.  Otherwise, return false.
315      * @throws AuthCompromisedException The client attempted to log in with an invalid rememberMe token.
316      */
317     protected function loginRememberedUser()
318     {
319         /** @var Birke\Rememberme\LoginResult $loginResult */
320         $loginResult = $this->rememberMe->login();
321 
322         if ($loginResult->isSuccess()) {
323             // Update in session
324             $this->session[$this->config['session.keys.current_user_id']] = $loginResult->getCredential();
325             // There is a chance that an attacker has stolen the login token,
326             // so we store the fact that the user was logged in via RememberMe (instead of login form)
327             $this->viaRemember = true;
328         } else {
329             // If $rememberMe->login() was not successfull, check if the token was invalid as well.  This means the cookie was stolen.
330             if ($loginResult->hasPossibleManipulation()) {
331                 throw new AuthCompromisedException();
332             }
333         }
334 
335         return $this->validateUserAccount($loginResult->getCredential());
336     }

338     /**
339      * Attempt to log in the client from the session.
340      *
341      * @return User|null If successful, the User object of the user in session.  Otherwise, return null.
342      * @throws AuthExpiredException The client attempted to use an expired rememberMe token.
343      */
344     protected function loginSessionUser()
345     {
346         $userId = $this->session->get($this->config['session.keys.current_user_id']);
347 
348         // If a user_id was found in the session, check any rememberMe cookie that was submitted.
349         // If they submitted an expired rememberMe cookie, then we need to log them out.
350         if ($userId) {
351             if (!$this->validateRememberMeCookie()) {
352                 $this->logout();
353                 throw new AuthExpiredException();
354             }
355         }
356 
357         return $this->validateUserAccount($userId);
358     }

360     /**
361      * Determine if the cookie contains a valid rememberMe token.
362      *
363      * @return bool
364      */
365     protected function validateRememberMeCookie()
366     {
367         $cookieValue = $this->rememberMe->getCookie()->getValue();
368         if (!$cookieValue) {
369             return true;
370         }
371         $triplet = RememberMeTriplet::fromString($cookieValue);
372         if (!$triplet->isValid()) {
373             return false;
374         }
375 
376         return true;
377     }

379     /**
380      * Tries to load the specified user by id from the database.
381      *
382      * Checks that the account is valid and enabled, throwing an exception if not.
383      * @param int $userId
384      * @return User|null
385      * @throws AccountInvalidException
386      * @throws AccountDisabledException
387      */
388     protected function validateUserAccount($userId)
389     {
390         if ($userId) {
391             $user = $this->classMapper->staticMethod('user', 'find', $userId);
392 
393             // If the user doesn't exist any more, throw an exception.
394             if (!$user) {
395                 throw new AccountInvalidException();
396             }
397 
398             // If the user has been disabled since their last request, throw an exception.
399             if (!$user->flag_enabled) {
400                 throw new AccountDisabledException();
401             }
402 
403             return $user;
404         } else {
405             return null;
406         }
