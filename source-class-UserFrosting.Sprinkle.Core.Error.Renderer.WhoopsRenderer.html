147                 $classes = [$class => $class] + class_parents($class) + class_implements($class);

165     public function render()
166     {
167         if (!$this->handleUnconditionally()) {
168             // Check conditions for outputting HTML:
169             // @todo: Make this more robust
170             if (php_sapi_name() === 'cli') {
171                 // Help users who have been relying on an internal test value
172                 // fix their code to the proper method
173                 if (isset($_ENV['whoops-test'])) {
174                     throw new \Exception(
175                         'Use handleUnconditionally instead of whoops-test'
176                         .' environment variable'
177                     );
178                 }
179 
180                 return Handler::DONE;
181             }
182         }
183 
184         $templateFile = $this->getResource("views/layout.html.php");
185         $cssFile      = $this->getResource("css/whoops.base.css");
186         $zeptoFile    = $this->getResource("js/zepto.min.js");
187         $clipboard    = $this->getResource("js/clipboard.min.js");
188         $jsFile       = $this->getResource("js/whoops.base.js");
189 
190         if ($this->customCss) {
191             $customCssFile = $this->getResource($this->customCss);
192         }
193 
194         $inspector = $this->getInspector();
195         $frames    = $inspector->getFrames();
196 
197         $code = $inspector->getException()->getCode();
198 
199         if ($inspector->getException() instanceof \ErrorException) {
200             // ErrorExceptions wrap the php-error types within the "severity" property
201             $code = Misc::translateErrorCode($inspector->getException()->getSeverity());
202         }
203 
204         // Detect frames that belong to the application.
205         if ($this->applicationPaths) {
206             /* @var \Whoops\Exception\Frame $frame */
207             foreach ($frames as $frame) {
208                 foreach ($this->applicationPaths as $path) {
209                     if (substr($frame->getFile(), 0, strlen($path)) === $path) {
210                         $frame->setApplication(true);
211                         break;
212                     }
213                 }
214             }
215         }
216 
217         // Nicely format the session object
218         $session = isset($_SESSION) ? $this->masked($_SESSION, '_SESSION') :  [];
219         $session = ['session' => Util::prettyPrintArray($session)];
220 
221         // List of variables that will be passed to the layout template.
222         $vars = [
223             "page_title" => $this->getPageTitle(),
224 
225             // @todo: Asset compiler
226             "stylesheet" => file_get_contents($cssFile),
227             "zepto"      => file_get_contents($zeptoFile),
228             "clipboard"  => file_get_contents($clipboard),
229             "javascript" => file_get_contents($jsFile),
230 
231             // Template paths:
232             "header"                     => $this->getResource("views/header.html.php"),
233             "header_outer"               => $this->getResource("views/header_outer.html.php"),

235             "frames_description"         => $this->getResource("views/frames_description.html.php"),
236             "frames_container"           => $this->getResource("views/frames_container.html.php"),

238             "panel_details_outer"        => $this->getResource("views/panel_details_outer.html.php"),
239             "panel_left"                 => $this->getResource("views/panel_left.html.php"),
240             "panel_left_outer"           => $this->getResource("views/panel_left_outer.html.php"),
241             "frame_code"                 => $this->getResource("views/frame_code.html.php"),
242             "env_details"                => $this->getResource("views/env_details.html.php"),
243 
244             "title"          => $this->getPageTitle(),
245             "name"           => explode("\\", $inspector->getExceptionName()),
246             "message"        => $inspector->getException()->getMessage(),
247             "code"           => $code,
248             "plain_exception" => Formatter::formatExceptionPlain($inspector),
249             "frames"         => $frames,
250             "has_frames"     => !!count($frames),
251             "handler"        => $this,
252             "handlers"       => [$this],
253 
254             "active_frames_tab" => count($frames) && $frames->offsetGet(0)->isApplication() ?  'application' : 'all',
255             "has_frames_tabs"   => $this->getApplicationPaths(),
256 
257             "tables"      => [
258                 "GET Data"              => $this->masked($_GET, '_GET'),
259                 "POST Data"             => $this->masked($_POST, '_POST'),
260                 "Files"                 => isset($_FILES) ? $this->masked($_FILES, '_FILES') : [],
261                 "Cookies"               => $this->masked($_COOKIE, '_COOKIE'),
262                 "Session"               => $session,
263                 "Server/Request Data"   => $this->masked($_SERVER, '_SERVER'),
264                 "Environment Variables" => $this->masked($_ENV, '_ENV'),
265             ],
266         ];
267 
268         if (isset($customCssFile)) {
269             $vars["stylesheet"] .= file_get_contents($customCssFile);
270         }
271 
272         // Add extra entries list of data tables:
273         // @todo: Consolidate addDataTable and addDataTableCallback
274         $extraTables = array_map(function ($table) use ($inspector) {
275             return $table instanceof \Closure ? $table($inspector) : $table;
276         }, $this->getDataTables());
277         $vars["tables"] = array_merge($extraTables, $vars["tables"]);
278

284         $this->templateHelper->setVariables($vars);
285 
286         ob_start();

289         $result = ob_get_clean();

591     /**
592      * Given a boolean if the editor link should
593      * act as an Ajax request. The editor must be a
594      * valid callable function/closure
595      *
596      * @param  string $filePath
597      * @param  int    $line
598      * @return array
599      */
600     protected function getEditor($filePath, $line)
601     {
602         if (!$this->editor || (!is_string($this->editor) && !is_callable($this->editor))) {
603             return [];
604         }
605 
606         if (is_string($this->editor) && isset($this->editors[$this->editor]) && !is_callable($this->editors[$this->editor])) {
607            return [
608                 'ajax' => false,
609                 'url' => $this->editors[$this->editor],
610             ];
611         }
612 
613         if (is_callable($this->editor) || (isset($this->editors[$this->editor]) && is_callable($this->editors[$this->editor]))) {
614             if (is_callable($this->editor)) {
615                 $callback = call_user_func($this->editor, $filePath, $line);
616             } else {
617                 $callback = call_user_func($this->editors[$this->editor], $filePath, $line);
618             }
619 
620             if (is_string($callback)) {
621                 return [
622                     'ajax' => false,
623                     'url' => $callback,
624                 ];
625             }
626 
627             return [
628                 'ajax' => isset($callback['ajax']) ? $callback['ajax'] : false,
629                 'url' => isset($callback['url']) ? $callback['url'] : $callback,
630             ];
631         }
632 
633         return [];
634     }

652     /**
653      * Finds a resource, by its relative path, in all available search paths.
654      * The search is performed starting at the last search path, and all the
655      * way back to the first, enabling a cascading-type system of overrides
656      * for all resources.
657      *
658      * @throws RuntimeException If resource cannot be found in any of the available paths
659      *
660      * @param  string $resource
661      * @return string
662      */
663     protected function getResource($resource)
664     {
665         // If the resource was found before, we can speed things up
666         // by caching its absolute, resolved path:
667         if (isset($this->resourceCache[$resource])) {
668             return $this->resourceCache[$resource];
669         }
670 
671         // Search through available search paths, until we find the
672         // resource we're after:
673         foreach ($this->searchPaths as $path) {
674             $fullPath = $path . "/$resource";
675 
676             if (is_file($fullPath)) {
677                 // Cache the result:
678                 $this->resourceCache[$resource] = $fullPath;
679                 return $fullPath;
680             }
681         }
682 
683         // If we got this far, nothing was found.
684         throw new RuntimeException(
685             "Could not find resource '$resource' in any resource paths."
686             . "(searched: " . join(", ", $this->searchPaths). ")"
687         );
688     }

690     /**
691      * Checks all values within the given superGlobal array.
692      * Blacklisted values will be replaced by a equal length string cointaining only '*' characters.
693      *
694      * We intentionally dont rely on $GLOBALS as it depends on 'auto_globals_jit' php.ini setting.
695      *
696      * @param array $superGlobal One of the superglobal arrays
697      * @param string $superGlobalName the name of the superglobal array, e.g. '_GET'
698      * @return array $values without sensitive data
699      */
700     private function masked(array $superGlobal, $superGlobalName)
701     {
702         $blacklisted = $this->blacklist[$superGlobalName];
703 
704         $values = $superGlobal;
705         foreach($blacklisted as $key) {
706             if (isset($superGlobal[$key])) {
707                 $values[$key] = str_repeat('*', strlen($superGlobal[$key]));
708             }
709         }
710         return $values;
