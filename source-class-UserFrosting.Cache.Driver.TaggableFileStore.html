 15 namespace UserFrosting\Cache\Driver;

 17 use Illuminate\Cache\FileStore;
 18 use Illuminate\Filesystem\Filesystem;
 19 
 20 class TaggableFileStore extends FileStore
 21 {
 22 
 23     /**
 24      * @var string The separator when creating a tagged directory.
 25      */
 26     public $separator;
 27 
 28     /**
 29      * @var string The directory where the tags list is stored
 30      */
 31     public $tagRepository = "cache_tags";
 32 
 33     /**
 34      * @var string Directory separator.
 35      */
 36     public $ds = "/";
 37 
 38     /**
 39      * Create a new file cache store instance.
 40      *
 41      * @param  \Illuminate\Filesystem\Filesystem $files
 42      * @param  string $directory
 43      * @param  array $options
 44      */
 45     public function __construct(Filesystem $files, $directory, $options)
 46     {
 47         $options = array_merge([
 48             'separator'=> '~#~'
 49         ], $options);
 50 
 51         $this->separator = $options['separator'];
 52         parent::__construct($files,$directory);
 53     }
 54 
 55 
 56     /**
 57      * Get the full path for the given cache key.
 58      *
 59      * @param  string  $key
 60      * @return string
 61      */
 62     protected function path($key)
 63     {
 64         $isTag = false;
 65         $split = explode($this->separator, $key);
 66 
 67         if(count($split) > 1)
 68         {
 69             $folder = reset($split) . $this->ds;
 70 
 71             if($folder === $this->tagRepository . $this->ds)
 72             {
 73                 $isTag = true;
 74             }
 75             $key = end($split);
 76         }
 77         else
 78         {
 79             $key = reset($split);
 80             $folder = '';
 81         }
 82 
 83         if($isTag)
 84         {
 85             $hash = $key;
 86             $parts = [];
 87         } else {
 88             $parts = array_slice(str_split($hash = sha1($key), 2), 0, 2);
 89         }
 90 
 91         return $this->directory . $this->ds . $folder . (count($parts) > 0 ? implode($this->ds, $parts) . $this->ds : '') . $hash;
 92     }
 93 
 94     /**
 95      * Begin executing a new tags operation.
 96      *
 97      * @param  array|mixed  $names
 98      * @return \Illuminate\Cache\TaggedCache
 99      */
100     public function tags($names)
101     {
102         return new TaggedFileCache($this, new FileTagSet($this, is_array($names) ? $names : func_get_args()));
103     }
104

106     /**
107      * Flush old tags path when a tag is flushed
108      *
109      * @param string $tagId
110      * @return void
111      */
112     public function flushOldTag($tagId){
113 
114         foreach ($this->files->directories($this->directory) as $directory)
115         {
116             if (str_contains(basename($directory),$tagId))
117             {
118                 $this->files->deleteDirectory($directory);
119             }
120         }
121     }
