 16 /**
 17  * Assets builder CLI Tools.
 18  * Wrapper for npm/node commands
 19  *
 20  * @extends Bakery
 21  * @author Alex Weissman (https://alexanderweissman.com)
 22  */
 23 class BuildAssets extends BaseCommand
 24 {
 25     /**
 26      * @var string Path to the build/ directory
 27      */
 28     protected $buildPath;
 29 
 30     /**
 31      * {@inheritDoc}
 32      */
 33     protected function configure()
 34     {
 35         $this->setName("build-assets")
 36              ->setDescription("Build the assets using node and npm")
 37              ->setHelp("The build directory contains the scripts and configuration files required to download Javascript, CSS, and other assets used by UserFrosting. This command will install Gulp, Bower, and several other required npm packages locally. With <info>npm</info> set up with all of its required packages, it can be use it to automatically download and install the assets in the correct directories. For more info, see <comment>https://learn.userfrosting.com/basics/installation</comment>")
 38              ->addOption("compile", "c", InputOption::VALUE_NONE, "Compile the assets and asset bundles for production environment")
 39              ->addOption("force", "f", InputOption::VALUE_NONE, "Force assets compilation by deleting cached data and installed assets before proceeding");
 40     }
 41 
 42     /**
 43      * {@inheritDoc}
 44      */
 45     protected function execute(InputInterface $input, OutputInterface $output)
 46     {
 47         // Display header,
 48         $this->io->title("UserFrosting's Assets Builder");
 49 
 50         // Set $path
 51         $this->buildPath = $this->projectRoot . \UserFrosting\DS . \UserFrosting\BUILD_DIR_NAME;
 52 
 53         // Delete cached data is requested
 54         if ($input->getOption('force')) {
 55             $this->clean();
 56         }
 57 
 58         // Perform tasks
 59         $this->npmInstall();
 60         $this->assetsInstall();
 61 
 62         // Compile if requested
 63         if ($input->getOption('compile') || $this->isProduction()) {
 64             $this->buildAssets();
 65         }
 66 
 67         // Test the result
 68         $this->checkAssets();
 69 
 70         // If all went well and there's no fatal errors, we are successful
 71         $this->io->success("Assets install looks successful");
 72     }
 73 
 74     /**
 75      * Install npm package
 76      *
 77      * @access protected
 78      * @return void
 79      */
 80     protected function npmInstall()
 81     {
 82         $this->io->section("<info>Installing npm dependencies</info>");
 83         $this->io->writeln("> <comment>npm install</comment>");
 84 
 85         // Temporarily change the working directory so we can install npm dependencies
 86         $wd = getcwd();
 87         chdir($this->buildPath);
 88         passthru("npm install");
 89         chdir($wd);
 90     }

 92     /**
 93      * Perform UF Assets installation

 98     protected function assetsInstall()

103     }

105     /**
106      * Build the production bundle.
107      *
108      * @access protected
109      * @return void
110      */
111     protected function buildAssets()
112     {
113         $this->io->section("Building assets for production");
114 
115         $this->io->writeln("> <comment>npm run uf-bundle-build</comment>");
116         passthru("npm run uf-bundle-build --prefix " . $this->buildPath);
117 
118         $this->io->writeln("> <comment>npm run uf-bundle</comment>");
119         passthru("npm run uf-bundle --prefix " . $this->buildPath);
120 
121         $this->io->writeln("> <comment>npm run uf-bundle-clean</comment>");
122         passthru("npm run uf-bundle-clean --prefix " . $this->buildPath);
123     }

125     /**
126      * Check that the assets where installed in the core sprinkles
127      *
128      * @access protected
129      * @return void
130      */
131     protected function checkAssets()
132     {
133         $this->io->section("Testing assets installation");
134 
135         // Get path and vendor files
136         $vendorPath = \UserFrosting\SPRINKLES_DIR . "/core/assets/vendor/*";
137         $coreVendorFiles = glob($vendorPath);
138 
139         if (!$coreVendorFiles){
140             $this->io->error("Assets installation seems to have failed. Directory `$vendorPath` is empty, but it shouldn't be. Check the above log for any errors.");
141             exit(1);
142         }
143 
144         // Check that `bundle.result.json` is present in production mode
145         $config = $this->ci->config;
146         $resultFile = \UserFrosting\ROOT_DIR . \UserFrosting\DS . \UserFrosting\BUILD_DIR_NAME . \UserFrosting\DS . $config['assets.compiled.schema'];
147         if ($this->isProduction() && !file_exists($resultFile)) {
148             $this->io->error("Assets building seems to have failed. File `$resultFile` not found. This file is required for production envrionement. Check the above log for any errors.");
149             exit(1);
150         }
151     }
