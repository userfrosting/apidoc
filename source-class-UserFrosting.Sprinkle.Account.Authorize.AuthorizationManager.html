 13 /**
 14  * AuthorizationManager class.
 15  *
 16  * Manages a collection of access condition callbacks, and uses them to perform access control checks on user objects.
 17  * @author Alex Weissman (https://alexanderweissman.com)
 18  */
 19 class AuthorizationManager
 20 {
 21     /**
 22      * @var ContainerInterface The global container object, which holds all your services.
 23      */
 24     protected $ci;
 25 
 26     /**
 27      * @var array[callable] An array of callbacks that accept some parameters and evaluate to true or false.
 28      */
 29     protected $callbacks = [];
 30 
 31     /**
 32      * Create a new AuthorizationManager object.
 33      *
 34      * @param ContainerInterface $ci The global container object, which holds all your services.
 35      */
 36     public function __construct(ContainerInterface $ci, array $callbacks = [])
 37     {
 38         $this->ci = $ci;
 39         $this->callbacks = $callbacks;
 40     }

 42     /**
 43      * Register an authorization callback, which can then be used in permission conditions.
 44      *
 45      * To add additional callbacks, simply extend the `authorizer` service in your Sprinkle's service provider.
 46      * @param string $name
 47      * @param callable $callback
 48      */
 49     public function addCallback($name, $callback)
 50     {
 51         $this->callbacks[$name] = $callback;
 52         return $this;
 53     }
 54 
 55     /**
 56      * Get all authorization callbacks.
 57      *
 58      * @return callable[]
 59      */
 60     public function getCallbacks()
 61     {
 62         return $this->callbacks;
 63     }

 65     /**
 66      * Checks whether or not a user has access on a particular permission slug.
 67      *
 68      * Determine if this user has access to the given $slug under the given $params.
 69      *
 70      * @param UserFrosting\Sprinkle\Account\Database\Models\User $user
 71      * @param string $slug The permission slug to check for access.
 72      * @param array $params[optional] An array of field names => values, specifying any additional data to provide the authorization module
 73      * when determining whether or not this user has access.
 74      * @return boolean True if the user has access, false otherwise.
 75      */
 76     public function checkAccess(User $user, $slug, array $params = [])
 77     {
 78         $debug = $this->ci->config['debug.auth'];
 79 
 80         if ($debug) {
 81             $trace = array_slice(debug_backtrace(DEBUG_BACKTRACE_IGNORE_ARGS, 3), 1);
 82             $this->ci->authLogger->debug("Authorization check requested at: ", $trace);
 83             $this->ci->authLogger->debug("Checking authorization for user {$user->id} ('{$user->user_name}') on permission '$slug'...");
 84         }
 85 
 86         if ($this->ci->authenticator->guest()) {
 87             if ($debug) {
 88                 $this->ci->authLogger->debug("User is not logged in.  Access denied.");
 89             }
 90             return false;
 91         }
 92 
 93         // The master (root) account has access to everything.
 94         // Need to use loose comparison for now, because some DBs return `id` as a string.
 95 
 96         if ($user->id == $this->ci->config['reserved_user_ids.master']) {
 97             if ($debug) {
 98                 $this->ci->authLogger->debug("User is the master (root) user.  Access granted.");
 99             }
100             return true;
101         }
102 
103         // Find all permissions that apply to this user (via roles), and check if any evaluate to true.
104         $permissions = $user->getCachedPermissions();
105 
106         if (empty($permissions) || !isset($permissions[$slug])) {
107             if ($debug) {
108                 $this->ci->authLogger->debug("No matching permissions found.  Access denied.");
109             }
110             return false;
111         }
112 
113         $permissions = $permissions[$slug];
114 
115         if ($debug) {
116             $this->ci->authLogger->debug("Found matching permissions: \n" . print_r($this->getPermissionsArrayDebugInfo($permissions), true));
117         }
118 
119         $nodeVisitor = new ParserNodeFunctionEvaluator($this->callbacks, $this->ci->authLogger, $debug);
120         $ace = new AccessConditionExpression($nodeVisitor, $user, $this->ci->authLogger, $debug);
121 
122         foreach ($permissions as $permission) {
123             $pass = $ace->evaluateCondition($permission->conditions, $params);
124             if ($pass) {
125                 if ($debug) {
126                     $this->ci->authLogger->debug("User passed conditions '{$permission->conditions}' .  Access granted.");
127                 }
128                 return true;
129             }
130         }
131 
132         if ($debug) {
133             $this->ci->authLogger->debug("User failed to pass any of the matched permissions.  Access denied.");
134         }
135 
136         return false;
137     }
138 
139     /**
140      * Remove extraneous information from the permission to reduce verbosity.
141      *
142      * @param  array
143      * @return array
144      */
145     protected function getPermissionsArrayDebugInfo($permissions)
146     {
147         $permissionsInfo = [];
148         foreach ($permissions as $permission) {
149             $permissionData = array_only($permission->toArray(), ['id', 'slug', 'name', 'conditions', 'description']);
150             // Remove this until we can find an efficient way to only load these once during debugging
151             //$permissionData['roles_via'] = $permission->roles_via->pluck('id')->all();
152             $permissionsInfo[] = $permissionData;
153         }
154 
155         return $permissionsInfo;
156     }
