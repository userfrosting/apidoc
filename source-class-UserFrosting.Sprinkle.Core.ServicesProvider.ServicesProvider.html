 60 /**
 61  * UserFrosting core services provider.
 62  *
 63  * Registers core services for UserFrosting, such as config, database, asset manager, translator, etc.
 64  * @author Alex Weissman (https://alexanderweissman.com)
 65  */
 66 class ServicesProvider
 67 {
 68     /**
 69      * Register UserFrosting's core services.
 70      *
 71      * @param ContainerInterface $container A DI container implementing ArrayAccess and container-interop.
 72      */
 73     public function register(ContainerInterface $container)
 74     {
 75         /**
 76          * Flash messaging service.
 77          *
 78          * Persists error/success messages between requests in the session.
 79          */
 80         $container['alerts'] = function ($c) {
 81             $config = $c->config;
 82 
 83             if ($config['alert.storage'] == 'cache') {
 84                 return new CacheAlertStream($config['alert.key'], $c->translator, $c->cache, $c->config);
 85             } elseif ($config['alert.storage'] == 'session') {
 86                 return new SessionAlertStream($config['alert.key'], $c->translator, $c->session);
 87             } else {
 88                 throw new \Exception("Bad alert storage handler type '{$config['alert.storage']}' specified in configuration file.");
 89             }
 90         };
 91 
 92         /**
 93          * Asset loader service.
 94          *
 95          * Loads assets from a specified relative location.
 96          * Assets are Javascript, CSS, image, and other files used by your site.
 97          */
 98         $container['assetLoader'] = function ($c) {
 99             $basePath = \UserFrosting\SPRINKLES_DIR;
100             $pattern = "/^[A-Za-z0-9_\-]+\/assets\//";
101 
102             $al = new AssetLoader($basePath, $pattern);
103             return $al;
104         };
105 
106         /**
107          * Asset manager service.
108          *
109          * Loads raw or compiled asset information from your bundle.config.json schema file.
110          * Assets are Javascript, CSS, image, and other files used by your site.
111          */
112         $container['assets'] = function ($c) {
113             $config = $c->config;
114             $locator = $c->locator;
115 
116             // Load asset schema
117             if ($config['assets.use_raw']) {
118                 $baseUrl = $config['site.uri.public'] . '/' . $config['assets.raw.path'];
119                 $removePrefix = \UserFrosting\APP_DIR_NAME . \UserFrosting\DS . \UserFrosting\SPRINKLES_DIR_NAME;
120                 $aub = new AssetUrlBuilder($locator, $baseUrl, $removePrefix, 'assets');
121 
122                 $as = new AssetBundleSchema($aub);
123 
124                 // Load Sprinkle assets
125                 $sprinkles = $c->sprinkleManager->getSprinkleNames();
126 
127                 // TODO: move this out into PathBuilder and Loader classes in userfrosting/assets
128                 // This would also allow us to define and load bundles in themes
129                 $bundleSchemas = array_reverse($locator->findResources('sprinkles://' . $config['assets.raw.schema'], true, true));
130 
131                 foreach ($bundleSchemas as $schema) {
132                     if (file_exists($schema)) {
133                         $as->loadRawSchemaFile($schema);
134                     }
135                 }
136             } else {
137                 $baseUrl = $config['site.uri.public'] . '/' . $config['assets.compiled.path'];
138                 $aub = new CompiledAssetUrlBuilder($baseUrl);
139 
140                 $as = new AssetBundleSchema($aub);
141                 $as->loadCompiledSchemaFile($locator->findResource("build://" . $config['assets.compiled.schema'], true, true));
142             }
143 
144             $am = new AssetManager($aub, $as);
145 
146             return $am;
147         };
148 
149         /**
150          * Cache service.
151          *
152          * @todo Create an option somewhere to flush the cache
153          */
154         $container['cache'] = function ($c) {
155 
156             $config = $c->config;
157 
158             if ($config['cache.driver'] == 'file') {
159                 $path = $c->locator->findResource('cache://', true, true);
160                 $cacheStore = new TaggableFileStore($path);
161             } elseif ($config['cache.driver'] == 'memcached') {
162                 $cacheStore = new MemcachedStore($config['cache.memcached']);
163             } elseif ($config['cache.driver'] == 'redis') {
164                 $cacheStore = new RedisStore($config['cache.redis']);
165             } else {
166                 throw new \Exception("Bad cache store type '{$config['cache.driver']}' specified in configuration file.");
167             }
168 
169             $cache = $cacheStore->instance();
170             return $cache->tags($config['cache.prefix']);
171         };

177          */
178         $container['checkEnvironment'] = function ($c) {
179             $checkEnvironment = new CheckEnvironment($c->view, $c->locator, $c->cache);
180             return $checkEnvironment;
181         };

183         /**
184          * Class mapper.
185          *
186          * Creates an abstraction on top of class names to allow extending them in sprinkles.
187          */
188         $container['classMapper'] = function ($c) {
189             $classMapper = new ClassMapper();
190             $classMapper->setClassMapping('query_builder', 'UserFrosting\Sprinkle\Core\Database\Builder');
191             $classMapper->setClassMapping('throttle', 'UserFrosting\Sprinkle\Core\Database\Models\Throttle');
192             return $classMapper;
193         };

195         /**
196          * Site config service (separate from Slim settings).
197          *
198          * Will attempt to automatically determine which config file(s) to use based on the value of the UF_MODE environment variable.
199          */
200         $container['config'] = function ($c) {
201             // Grab any relevant dotenv variables from the .env file
202             try {
203                 $dotenv = new Dotenv(\UserFrosting\APP_DIR);
204                 $dotenv->load();
205             } catch (InvalidPathException $e) {
206                 // Skip loading the environment config file if it doesn't exist.
207             }
208 
209             // Get configuration mode from environment
210             $mode = getenv('UF_MODE') ?: '';
211 
212             // Construct and load config repository
213             $builder = new ConfigPathBuilder($c->locator, 'config://');
214             $loader = new ArrayFileLoader($builder->buildPaths($mode));
215             $config = new Repository($loader->load());
216 
217             // Construct base url from components, if not explicitly specified
218             if (!isset($config['site.uri.public'])) {
219                 $base_uri = $config['site.uri.base'];
220 
221                 $public = new Uri(
222                     $base_uri['scheme'],
223                     $base_uri['host'],
224                     $base_uri['port'],
225                     $base_uri['path']
226                 );
227 
228                 // Slim\Http\Uri likes to add trailing slashes when the path is empty, so this fixes that.
229                 $config['site.uri.public'] = trim($public, '/');
230             }
231 
232             // Hacky fix to prevent sessions from being hit too much: ignore CSRF middleware for requests for raw assets ;-)
233             // See https://github.com/laravel/framework/issues/8172#issuecomment-99112012 for more information on why it's bad to hit Laravel sessions multiple times in rapid succession.
234             $csrfBlacklist = $config['csrf.blacklist'];
235             $csrfBlacklist['^/' . $config['assets.raw.path']] = [
236                 'GET'
237             ];
238 
239             $config->set('csrf.blacklist', $csrfBlacklist);
240 
241             return $config;
242         };

244         /**
245          * Initialize CSRF guard middleware.
246          *
247          * @see https://github.com/slimphp/Slim-Csrf
248          */
249         $container['csrf'] = function ($c) {
250             $csrfKey = $c->config['session.keys.csrf'];
251 
252             // Workaround so that we can pass storage into CSRF guard.
253             // If we tried to directly pass the indexed portion of `session` (for example, $c->session['site.csrf']),
254             // we would get an 'Indirect modification of overloaded element of UserFrosting\Session\Session' error.
255             // If we tried to assign an array and use that, PHP would only modify the local variable, and not the session.
256             // Since ArrayObject is an object, PHP will modify the object itself, allowing it to persist in the session.
257             if (!$c->session->has($csrfKey)) {
258                 $c->session[$csrfKey] = new \ArrayObject();
259             }
260             $csrfStorage = $c->session[$csrfKey];
261 
262             $onFailure = function ($request, $response, $next) {
263                 $e = new BadRequestException("The CSRF code was invalid or not provided.");
264                 $e->addUserMessage('CSRF_MISSING');
265                 throw $e;
266 
267                 return $next($request, $response);
268             };
269 
270             return new Guard($c->config['csrf.name'], $csrfStorage, $onFailure, $c->config['csrf.storage_limit'], $c->config['csrf.strength'], $c->config['csrf.persistent_token']);
271         };
