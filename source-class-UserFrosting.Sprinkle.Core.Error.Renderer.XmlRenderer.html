13 class XmlRenderer extends ErrorRenderer

15     /**
16      * @return string
17      */
18     public function render()
19     {
20         $e = $this->exception;
21         $xml = "<error>\n  <message>UserFrosting Application Error</message>\n";
22         if ($this->displayErrorDetails) {
23             do {
24                 $xml .= "  <exception>\n";
25                 $xml .= "    <type>" . get_class($e) . "</type>\n";
26                 $xml .= "    <code>" . $e->getCode() . "</code>\n";
27                 $xml .= "    <message>" . $this->createCdataSection($e->getMessage()) . "</message>\n";
28                 $xml .= "    <file>" . $e->getFile() . "</file>\n";
29                 $xml .= "    <line>" . $e->getLine() . "</line>\n";
30                 $xml .= "  </exception>\n";
31             } while ($e = $e->getPrevious());
32         }
33         $xml .= "</error>";
34 
35         return $xml;
36     }
37 
38     /**
39      * Returns a CDATA section with the given content.
40      *
41      * @param  string $content
42      * @return string
43      */
44     private function createCdataSection($content)
45     {
46         return sprintf('<![CDATA[%s]]>', str_replace(']]>', ']]]]><![CDATA[>', $content));
47     }
