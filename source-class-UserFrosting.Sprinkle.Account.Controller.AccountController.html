  46      * Check a username for availability.
  47      *
  48      * This route is throttled by default, to discourage abusing it for account enumeration.
  49      * This route is "public access".
  50      * Request type: GET
  51      */
  52     public function checkUsername($request, $response, $args)
  53     {
  54         /** @var UserFrosting\Sprinkle\Core\MessageStream $ms */
  55         $ms = $this->ci->alerts;
  56 
  57         // GET parameters
  58         $params = $request->getQueryParams();
  59 
  60         // Load request schema
  61         $schema = new RequestSchema('schema://requests/check-username.yaml');
  62 
  63         // Whitelist and set parameter defaults
  64         $transformer = new RequestDataTransformer($schema);
  65         $data = $transformer->transform($params);
  66 
  67         // Validate, and halt on validation errors.
  68         $validator = new ServerSideValidator($schema, $this->ci->translator);

  70             // TODO: encapsulate the communication of error messages from ServerSideValidator to the BadRequestException

  81         $throttler = $this->ci->throttler;
  82         $delay = $throttler->getDelay('check_username_request');
  83 
  84         // Throttle requests
  85         if ($delay > 0) {
  86             return $response->withStatus(429);
  87         }
  88 
  89         /** @var UserFrosting\Sprinkle\Core\Util\ClassMapper $classMapper */
  90         $classMapper = $this->ci->classMapper;

  93         $translator = $this->ci->translator;

  95         // Log throttleable event
  96         $throttler->logEvent('check_username_request');

 106     /**
 107      * Processes a request to cancel a password reset request.
 108      *
 109      * This is provided so that users can cancel a password reset request, if they made it in error or if it was not initiated by themselves.
 110      * Processes the request from the password reset link, checking that:
 111      * 1. The provided token is associated with an existing user account, who has a pending password reset request.
 112      * Request type: GET
 113      */
 114     public function denyResetPassword($request, $response, $args)
 115     {
 116         // GET parameters
 117         $params = $request->getQueryParams();
 118 
 119         /** @var UserFrosting\Sprinkle\Core\MessageStream $ms */
 120         $ms = $this->ci->alerts;
 121 
 122         /** @var UserFrosting\Sprinkle\Core\Util\ClassMapper $classMapper */
 123         $classMapper = $this->ci->classMapper;
 124 
 125         $loginPage = $this->ci->router->pathFor('login');
 126 
 127         // Load validation rules
 128         $schema = new RequestSchema('schema://requests/deny-password.yaml');
 129 
 130         // Whitelist and set parameter defaults
 131         $transformer = new RequestDataTransformer($schema);
 132         $data = $transformer->transform($params);
 133 
 134         // Validate, and halt on validation errors.  Since this is a GET request, we need to redirect on failure
 135         $validator = new ServerSideValidator($schema, $this->ci->translator);
 136         if (!$validator->validate($data)) {
 137             $ms->addValidationErrors($validator);
 138             // 400 code + redirect is perfectly fine, according to user Dilaz in #laravel
 139             return $response->withRedirect($loginPage, 400);
 140         }
 141 
 142         $passwordReset = $this->ci->repoPasswordReset->cancel($data['token']);
 143 
 144         if (!$passwordReset) {
 145             $ms->addMessageTranslated('danger', 'PASSWORD.FORGET.INVALID');
 146             return $response->withRedirect($loginPage, 400);
 147         }
 148 
 149         $ms->addMessageTranslated('success', 'PASSWORD.FORGET.REQUEST_CANNED');
 150         return $response->withRedirect($loginPage);
 151     }

 153     /**
 154      * Processes a request to email a forgotten password reset link to the user.
 155      *
 156      * Processes the request from the form on the "forgot password" page, checking that:
 157      * 1. The rate limit for this type of request is being observed.
 158      * 2. The provided email address belongs to a registered account;
 159      * 3. The submitted data is valid.
 160      * Note that we have removed the requirement that a password reset request not already be in progress.
 161      * This is because we need to allow users to re-request a reset, even if they lose the first reset email.
 162      * This route is "public access".
 163      * Request type: POST
 164      * @todo require additional user information
 165      * @todo prevent password reset requests for root account?
 166      */
 167     public function forgotPassword($request, $response, $args)
 168     {
 169         /** @var UserFrosting\Sprinkle\Core\MessageStream $ms */
 170         $ms = $this->ci->alerts;
 171 
 172         /** @var UserFrosting\Sprinkle\Core\Util\ClassMapper $classMapper */
 173         $classMapper = $this->ci->classMapper;
 174 
 175         /** @var UserFrosting\Config\Config $config */
 176         $config = $this->ci->config;
 177 
 178         // Get POST parameters
 179         $params = $request->getParsedBody();
 180 
 181         // Load the request schema
 182         $schema = new RequestSchema('schema://requests/forgot-password.yaml');
 183 
 184         // Whitelist and set parameter defaults
 185         $transformer = new RequestDataTransformer($schema);
 186         $data = $transformer->transform($params);
 187 
 188         // Validate, and halt on validation errors.  Failed validation attempts do not count towards throttling limit.
 189         $validator = new ServerSideValidator($schema, $this->ci->translator);
 190         if (!$validator->validate($data)) {
 191             $ms->addValidationErrors($validator);
 192             return $response->withStatus(400);
 193         }
 194 
 195         // Throttle requests
 196 
 197         /** @var UserFrosting\Sprinkle\Core\Throttle\Throttler $throttler */
 198         $throttler = $this->ci->throttler;
 199 
 200         $throttleData = [
 201             'email' => $data['email']
 202         ];
 203         $delay = $throttler->getDelay('password_reset_request', $throttleData);
 204 
 205         if ($delay > 0) {
 206             $ms->addMessageTranslated('danger', 'RATE_LIMIT_EXCEEDED', ['delay' => $delay]);
 207             return $response->withStatus(429);
 208         }
 209 
 210         // All checks passed!  log events/activities, update user, and send email
 211         // Begin transaction - DB will be rolled back if an exception occurs
 212         Capsule::transaction( function() use ($classMapper, $data, $throttler, $throttleData, $config) {
 213 
 214             // Log throttleable event
 215             $throttler->logEvent('password_reset_request', $throttleData);
 216 
 217             // Load the user, by email address
 218             $user = $classMapper->staticMethod('user', 'where', 'email', $data['email'])->first();
 219 
 220             // Check that the email exists.
 221             // If there is no user with that email address, we should still pretend like we succeeded, to prevent account enumeration
 222             if ($user) {
 223                 // Try to generate a new password reset request.
 224                 // Use timeout for "reset password"
 225                 $passwordReset = $this->ci->repoPasswordReset->create($user, $config['password_reset.timeouts.reset']);
 226 
 227                 // Create and send email
 228                 $message = new TwigMailMessage($this->ci->view, 'mail/password-reset.html.twig');
 229                 $message->from($config['address_book.admin'])
 230                         ->addEmailRecipient(new EmailRecipient($user->email, $user->full_name))
 231                         ->addParams([
 232                             'user' => $user,
 233                             'token' => $passwordReset->getToken(),
 234                             'request_date' => Carbon::now()->format('Y-m-d H:i:s')
 235                         ]);
 236 
 237                 $this->ci->mailer->send($message);
 238             }
 239         });
 240 
 241         // TODO: create delay to prevent timing-based attacks
 242 
 243         $ms->addMessageTranslated('success', 'PASSWORD.FORGET.REQUEST_SENT', ['email' => $data['email']]);
 244         return $response->withStatus(200);
 245     }

 273     /**
 274      * Processes an account login request.
 275      *
 276      * Processes the request from the form on the login page, checking that:
 277      * 1. The user is not already logged in.
 278      * 2. The rate limit for this type of request is being observed.
 279      * 3. Email login is enabled, if an email address was used.
 280      * 4. The user account exists.
 281      * 5. The user account is enabled and verified.
 282      * 6. The user entered a valid username/email and password.
 283      * This route, by definition, is "public access".
 284      * Request type: POST
 285      */
 286     public function login($request, $response, $args)
 287     {
 288         /** @var UserFrosting\Sprinkle\Core\MessageStream $ms */
 289         $ms = $this->ci->alerts;
 290 
 291         /** @var UserFrosting\Sprinkle\Account\Database\Models\User $currentUser */
 292         $currentUser = $this->ci->currentUser;
 293 
 294         /** @var UserFrosting\Sprinkle\Account\Authenticate\Authenticator $authenticator */
 295         $authenticator = $this->ci->authenticator;
 296 
 297         // Return 200 success if user is already logged in
 298         if ($authenticator->check()) {
 299             $ms->addMessageTranslated('warning', 'LOGIN.ALREADY_COMPLETE');
 300             return $response->withStatus(200);
 301         }
 302 
 303         /** @var UserFrosting\Config\Config $config */
 304         $config = $this->ci->config;
 305 
 306         // Get POST parameters
 307         $params = $request->getParsedBody();
 308 
 309         // Load the request schema
 310         $schema = new RequestSchema('schema://requests/login.yaml');
 311 
 312         // Whitelist and set parameter defaults
 313         $transformer = new RequestDataTransformer($schema);
 314         $data = $transformer->transform($params);
 315 
 316         // Validate, and halt on validation errors.  Failed validation attempts do not count towards throttling limit.
 317         $validator = new ServerSideValidator($schema, $this->ci->translator);
 318         if (!$validator->validate($data)) {
 319             $ms->addValidationErrors($validator);
 320             return $response->withStatus(400);
 321         }
 322 
 323         // Determine whether we are trying to log in with an email address or a username
 324         $isEmail = filter_var($data['user_name'], FILTER_VALIDATE_EMAIL);
 325 
 326         // Throttle requests
 327 
 328         /** @var UserFrosting\Sprinkle\Core\Throttle\Throttler $throttler */
 329         $throttler = $this->ci->throttler;
 330 
 331         $userIdentifier = $data['user_name'];
 332 
 333         $throttleData = [
 334             'user_identifier' => $userIdentifier
 335         ];
 336 
 337         $delay = $throttler->getDelay('sign_in_attempt', $throttleData);
 338         if ($delay > 0) {
 339             $ms->addMessageTranslated('danger', 'RATE_LIMIT_EXCEEDED', [
 340                 'delay' => $delay
 341             ]);
 342             return $response->withStatus(429);
 343         }
 344 
 345         // Log throttleable event
 346         $throttler->logEvent('sign_in_attempt', $throttleData);
 347 
 348         // If credential is an email address, but email login is not enabled, raise an error.
 349         // Note that we do this after logging throttle event, so this error counts towards throttling limit.
 350         if ($isEmail && !$config['site.login.enable_email']) {
 351             $ms->addMessageTranslated('danger', 'USER_OR_PASS_INVALID');
 352             return $response->withStatus(403);
 353         }
 354 
 355         // Try to authenticate the user.  Authenticator will throw an exception on failure.
 356         /** @var UserFrosting\Sprinkle\Account\Authenticate\Authenticator $authenticator */
 357         $authenticator = $this->ci->authenticator;
 358 
 359         $currentUser = $authenticator->attempt(($isEmail ? 'email' : 'user_name'), $userIdentifier, $data['password'], $data['rememberme']);
 360 
 361         $ms->addMessageTranslated('success', 'WELCOME', $currentUser->export());
 362 
 363         // Set redirect, if relevant
 364         $redirectOnLogin = $this->ci->get('redirect.onLogin');
 365 
 366         return $redirectOnLogin($request, $response, $args);
 367     }

 678     /**
 679      * Processes an new account registration request.
 680      *
 681      * This is throttled to prevent account enumeration, since it needs to divulge when a username/email has been used.
 682      * Processes the request from the form on the registration page, checking that:
 683      * 1. The honeypot was not modified;
 684      * 2. The master account has already been created (during installation);
 685      * 3. Account registration is enabled;
 686      * 4. The user is not already logged in;
 687      * 5. Valid information was entered;
 688      * 6. The captcha, if enabled, is correct;
 689      * 7. The username and email are not already taken.
 690      * Automatically sends an activation link upon success, if account activation is enabled.
 691      * This route is "public access".
 692      * Request type: POST
 693      * Returns the User Object for the user record that was created.
 694      */
 695     public function register(Request $request, Response $response, $args)
 696     {
 697         /** @var MessageStream $ms */
 698         $ms = $this->ci->alerts;
 699 
 700         /** @var UserFrosting\Sprinkle\Core\Util\ClassMapper $classMapper */
 701         $classMapper = $this->ci->classMapper;
 702 
 703         /** @var UserFrosting\Config\Config $config */
 704         $config = $this->ci->config;
 705 
 706         // Get POST parameters: user_name, first_name, last_name, email, password, passwordc, captcha, spiderbro, csrf_token
 707         $params = $request->getParsedBody();
 708 
 709         // Check the honeypot. 'spiderbro' is not a real field, it is hidden on the main page and must be submitted with its default value for this to be processed.
 710         if (!isset($params['spiderbro']) || $params['spiderbro'] != 'http://') {
 711             throw new SpammyRequestException('Possible spam received:' . print_r($params, true));
 712         }
 713 
 714         // Security measure: do not allow registering new users until the master account has been created.
 715         if (!$classMapper->staticMethod('user', 'find', $config['reserved_user_ids.master'])) {
 716             $ms->addMessageTranslated('danger', 'ACCOUNT.MASTER_NOT_EXISTS');
 717             return $response->withStatus(403);
 718         }
 719 
 720         // Check if registration is currently enabled
 721         if (!$config['site.registration.enabled']) {
 722             $ms->addMessageTranslated('danger', 'REGISTRATION.DISABLED');
 723             return $response->withStatus(403);
 724         }
 725 
 726         /** @var UserFrosting\Sprinkle\Account\Authenticate\Authenticator $authenticator */
 727         $authenticator = $this->ci->authenticator;
 728 
 729         // Prevent the user from registering if he/she is already logged in
 730         if ($authenticator->check()) {
 731             $ms->addMessageTranslated('danger', 'REGISTRATION.LOGOUT');
 732             return $response->withStatus(403);
 733         }
 734 
 735         // Load the request schema
 736         $schema = new RequestSchema('schema://requests/register.yaml');
 737 
 738         // Whitelist and set parameter defaults
 739         $transformer = new RequestDataTransformer($schema);
 740         $data = $transformer->transform($params);
 741 
 742         $error = false;
 743 
 744         // Validate request data
 745         $validator = new ServerSideValidator($schema, $this->ci->translator);
 746         if (!$validator->validate($data)) {
 747             $ms->addValidationErrors($validator);
 748             $error = true;
 749         }
 750 
 751         /** @var UserFrosting\Sprinkle\Core\Throttle\Throttler $throttler */
 752         $throttler = $this->ci->throttler;
 753         $delay = $throttler->getDelay('registration_attempt');
 754 
 755         // Throttle requests
 756         if ($delay > 0) {
 757             return $response->withStatus(429);
 758         }
 759 
 760         // Check if username or email already exists
 761         if ($classMapper->staticMethod('user', 'findUnique', $data['user_name'], 'user_name')) {
 762             $ms->addMessageTranslated('danger', 'USERNAME.IN_USE', $data);
 763             $error = true;
 764         }
 765 
 766         if ($classMapper->staticMethod('user', 'findUnique', $data['email'], 'email')) {
 767             $ms->addMessageTranslated('danger', 'EMAIL.IN_USE', $data);
 768             $error = true;
 769         }
 770 
 771         // Check captcha, if required
 772         if ($config['site.registration.captcha']) {
 773             $captcha = new Captcha($this->ci->session, $this->ci->config['session.keys.captcha']);
 774             if (!$data['captcha'] || !$captcha->verifyCode($data['captcha'])) {
 775                 $ms->addMessageTranslated('danger', 'CAPTCHA.FAIL');
 776                 $error = true;
 777             }
 778         }
 779 
 780         if ($error) {
 781             return $response->withStatus(400);
 782         }
 783 
 784         // Remove captcha, password confirmation from object data after validation
 785         unset($data['captcha']);
 786         unset($data['passwordc']);
 787 
 788         if ($config['site.registration.require_email_verification']) {
 789             $data['flag_verified'] = false;
 790         } else {
 791             $data['flag_verified'] = true;
 792         }
 793 
 794         // Load default group
 795         $groupSlug = $config['site.registration.user_defaults.group'];
 796         $defaultGroup = $classMapper->staticMethod('group', 'where', 'slug', $groupSlug)->first();
 797 
 798         if (!$defaultGroup) {
 799             $e = new HttpException("Account registration is not working because the default group '$groupSlug' does not exist.");
 800             $e->addUserMessage('ACCOUNT.REGISTRATION_BROKEN');
 801             throw $e;
 802         }
 803 
 804         // Set default group
 805         $data['group_id'] = $defaultGroup->id;
 806 
 807         // Set default locale
 808         $data['locale'] = $config['site.registration.user_defaults.locale'];
 809 
 810         // Hash password
 811         $data['password'] = Password::hash($data['password']);
 812 
 813         // All checks passed!  log events/activities, create user, and send verification email (if required)
 814         // Begin transaction - DB will be rolled back if an exception occurs
 815         Capsule::transaction( function() use ($classMapper, $data, $ms, $config, $throttler) {
 816             // Log throttleable event
 817             $throttler->logEvent('registration_attempt');
 818 
 819             // Create the user
 820             $user = $classMapper->createInstance('user', $data);
 821 
 822             // Store new user to database
 823             $user->save();
 824 
 825             // Create activity record
 826             $this->ci->userActivityLogger->info("User {$user->user_name} registered for a new account.", [
 827                 'type' => 'sign_up',
 828                 'user_id' => $user->id
 829             ]);
 830 
 831             // Load default roles
 832             $defaultRoleSlugs = $classMapper->staticMethod('role', 'getDefaultSlugs');
 833             $defaultRoles = $classMapper->staticMethod('role', 'whereIn', 'slug', $defaultRoleSlugs)->get();
 834             $defaultRoleIds = $defaultRoles->pluck('id')->all();
 835 
 836             // Attach default roles
 837             $user->roles()->attach($defaultRoleIds);
 838 
 839             // Verification email
 840             if ($config['site.registration.require_email_verification']) {
 841                 // Try to generate a new verification request
 842                 $verification = $this->ci->repoVerification->create($user, $config['verification.timeout']);
 843 
 844                 // Create and send verification email
 845                 $message = new TwigMailMessage($this->ci->view, 'mail/verify-account.html.twig');
 846 
 847                 $message->from($config['address_book.admin'])
 848                         ->addEmailRecipient(new EmailRecipient($user->email, $user->full_name))
 849                         ->addParams([
 850                             'user' => $user,
 851                             'token' => $verification->getToken()
 852                         ]);
 853 
 854                 $this->ci->mailer->send($message);
 855 
 856                 $ms->addMessageTranslated('success', 'REGISTRATION.COMPLETE_TYPE2', $user->toArray());
 857             } else {
 858                 // No verification required
 859                 $ms->addMessageTranslated('success', 'REGISTRATION.COMPLETE_TYPE1');
 860             }
 861         });
 862 
 863         return $response->withStatus(200);
 864     }
