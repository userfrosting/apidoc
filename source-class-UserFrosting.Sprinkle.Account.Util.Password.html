10 /**
11  * Password utility class
12  *
13  * @author Alex Weissman (https://alexanderweissman.com)
14  */
15 class Password
16 {
17     /**
18      * Returns the hashing type for a specified password hash.
19      *
20      * Automatically detects the hash type: "sha1" (for UserCake legacy accounts), "legacy" (for 0.1.x accounts), and "modern" (used for new accounts).
21      * @param string $password the hashed password.
22      * @return string "sha1"|"legacy"|"modern".
23      */
24     public static function getHashType($password)
25     {
26         // If the password in the db is 65 characters long, we have an sha1-hashed password.
27         if (strlen($password) == 65) {
28             return 'sha1';
29         } elseif (substr($password, 0, 7) == '$2y$12$') {
30             return 'legacy';
31         }
32 
33         return 'modern';
34     }
35 
36     /**
37      * Hashes a plaintext password using bcrypt.
38      *
39      * @param string $password the plaintext password.
40      * @return string the hashed password.
41      * @throws HashFailedException
42      */
43     public static function hash($password)
44     {
45         $hash = password_hash($password, PASSWORD_BCRYPT);
46 
47         if (!$hash) {
48             throw new HashFailedException();
49         }
50 
51         return $hash;
52     }
53 
54     /**
55      * Verify a plaintext password against the user's hashed password.
56      *
57      * @param string $password The plaintext password to verify.
58      * @param string $hash The hash to compare against.
59      * @return boolean True if the password matches, false otherwise.
60      */
61     public static function verify($password, $hash)
62     {
63         if (static::getHashType($hash) == 'sha1') {
64             // Legacy UserCake passwords
65             $salt = substr($hash, 0, 25);       // Extract the salt from the hash
66             $hashInput = $salt . sha1($salt . $password);
67             if ($hashInput == $hash) {
68                 return true;
69             }
70 
71             return false;
72 
73         } elseif (static::getHashType($hash) == 'legacy') {
74             // Homegrown implementation (assuming that current install has been using a cost parameter of 12)
75             // Used for manual implementation of bcrypt.
76             $cost = '12';
77             if (substr($hash, 0, 60) == crypt($password, '$2y$' . $cost . '$' . substr($hash, 60))) {
78                 return true;
79             }
80 
81             return false;
82         }
83 
84         // Modern implementation
85         return password_verify($password, $hash);
86     }
