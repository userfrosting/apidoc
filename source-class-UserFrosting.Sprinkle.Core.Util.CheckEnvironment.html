 59     /**
 60      * Invoke the CheckEnvironment middleware, performing all pre-flight checks and returning an error page if problems were found.
 61      *
 62      * @param  \Psr\Http\Message\ServerRequestInterface $request  PSR7 request
 63      * @param  \Psr\Http\Message\ResponseInterface      $response PSR7 response
 64      * @param  callable                                 $next     Next middleware
 65      *
 66      * @return \Psr\Http\Message\ResponseInterface
 67      */
 68     public function __invoke($request, $response, $next)
 69     {
 70         $problemsFound = false;
 71 
 72         // If production environment and no cached checks, perform environment checks
 73         if ($this->isProduction() && $this->cache->get('checkEnvironment') != 'pass') {
 74             $problemsFound = $this->checkAll();
 75 
 76             // Cache if checks passed
 77             if (!$problemsFound) {
 78                 $this->cache->forever('checkEnvironment', 'pass');
 79             }
 80         } elseif (!$this->isProduction()) {
 81             $problemsFound = $this->checkAll();
 82         }
 83 
 84         if ($problemsFound) {
 85             $results = array_merge($this->resultsFailed, $this->resultsSuccess);
 86 
 87             $response = $this->view->render($response, 'pages/error/config-errors.html.twig', [
 88                 "messages" => $results
 89             ]);
 90         } else {
 91             $response = $next($request, $response);
 92         }
 93 
 94         return $response;
 95     }

245     /**
246      * Check that log, cache, and session directories are writable, and that other directories are set appropriately for the environment.
247      */
248     function checkPermissions()
249     {
250         $problemsFound = false;
251 
252         $shouldBeWriteable = [
253             $this->locator->findResource('log://') => true,
254             $this->locator->findResource('cache://') => true,
255             $this->locator->findResource('session://') => true
256         ];
257 
258         if ($this->isProduction()) {
259             // Should be write-protected in production!
260             $shouldBeWriteable = array_merge($shouldBeWriteable, [
261                 \UserFrosting\SPRINKLES_DIR => false,
262                 \UserFrosting\VENDOR_DIR => false
263             ]);
264         }
265 
266         // Check for essential files & perms
267         foreach ($shouldBeWriteable as $file => $assertWriteable) {
268             $is_dir = false;
269             if (!file_exists($file)) {
270                 $problemsFound = true;
271                 $this->resultsFailed['file-' . $file] = [
272                     "title" => "<i class='fa fa-file-o fa-fw'></i> File or directory does not exist.",
273                     "message" => "We could not find the file or directory <code>$file</code>.",
274                     "success" => false
275                 ];
276             } else {
277                 $writeable = is_writable($file);
278                 if ($assertWriteable !== $writeable) {
279                     $problemsFound = true;
280                     $this->resultsFailed['file-' . $file] = [
281                         "title" => "<i class='fa fa-file-o fa-fw'></i> Incorrect permissions for file or directory.",
282                         "message" => "<code>$file</code> is "
283                             . ($writeable ? "writeable" : "not writeable")
284                             . ", but it should "
285                             . ($assertWriteable ? "be writeable" : "not be writeable")
286                             . ".  Please modify the OS user or group permissions so that user <b>"
287                             . exec('whoami') . "</b> "
288                             . ($assertWriteable ? "has" : "does not have") . " write permissions for this directory.",
289                         "success" => false
290                     ];
291                 } else {
292                     $this->resultsSuccess['file-' . $file] = [
293                         "title" => "<i class='fa fa-file-o fa-fw'></i> File/directory check passed!",
294                         "message" => "<code>$file</code> exists and is correctly set as <b>"
295                             . ($writeable ? "writeable" : "not writeable")
296                             . "</b>.",
297                         "success" => true
298                     ];
299                 }
300             }
301         }
302         return $problemsFound;
303     }
